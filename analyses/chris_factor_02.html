def rank1_approx(a):
    u, s, v = nla.svd(a, False)
    return s[0] * np.outer(u[:, 0], v[0, :])

def rank2_approx(a):
    u, s, v = nla.svd(a, False)
    return s[0] * np.outer(u[:, 0], v[0, :]) + s[1] * np.outer(u[:, 1], v[1, :])

def rel_mse(y1, y2):    
    yh1, yh2 = fitted(y1, y2)
    return np.hstack([np.sum((yh1 - y1)**2), np.sum((yh2 - y2)**2)])

def fitted(y1, y2):    
    bt1 = spo.nnls(amat1, y1)[0]
    bt2 = spo.nnls(amat2, y2)[0]   
    yh1 = sdot(amat1, bt1)
    yh2 = sdot(amat2, bt2)
    return yh1, yh2

def defactor1(xx):
    dat = xx[:, 3:]
    ys1 = dat[:, diff_inds_1]
    ys2 = dat[:, diff_inds_2]
    yhs1 = np.vstack([fitted1(y) for y in ys1])
    yhs2 = np.vstack([fitted2(y) for y in ys2])
    resids1 = ys1 - yhs1
    resids2 = ys2 - yhs2    
    rresids1 = rank1_approx(resids1)
    rresids2 = rank1_approx(resids2)
    ys1_adj = ys1 - rresids1
    ys2_adj = ys2 - rresids2    
    yhs1_new = np.vstack([fitted1(y) for y in ys1_adj])
    yhs2_new = np.vstack([fitted2(y) for y in ys2_adj])
    emd_errs = np.array([rel_emd(y1, y2) for (y1, y2) in zip(ys1_adj, ys2_adj)])
    c_resids1 = ys2 - yhs1_new
    c_resids2 = ys1 - yhs2_new
    return np.hstack([sum(emd_errs), sum(c_resids1.ravel()**2), sum(c_resids2.ravel()**2)])
